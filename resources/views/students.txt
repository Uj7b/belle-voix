function initials(n) {

function colorFor(n) {
  if (!n) return COLORS[0];
  var h = 0;
  for (var i = 0; i < n.length; i++) h = (h * 31 + n.charCodeAt(i)) & 0xffff;
  return COLORS[h % COLORS.length];
}

}

function filterTable() {
  var q   = document.getElementById('searchInput').value.toLowerCase();
  var cls = document.getElementById('classFilter').value;
  var st  = document.getElementById('statusFilter').value;
  var gn  = document.getElementById('genderFilter').value;

  filtered = students.filter(function(s) {
    var name  = s.user && s.user.fullname   ? s.user.fullname.toLowerCase()  : '';
    var cin   = s.user && s.user.cin        ? s.user.cin.toLowerCase()        : '';
    var scls  = s.school_class             ? s.school_class                   : null;
    var sgend = s.user && s.user.gender    ? s.user.gender                    : '';

    return (!q  || name.includes(q) || cin.includes(q)) &&
           (!cls || (scls && (scls.name == cls || String(scls.id) == cls))) &&
           (!st  || s.status == st) &&
           (!gn  || sgend == gn);
  });
  page = 1;
  render();
}

function sortTable() {
  var v = document.getElementById('sortBy').value;
  filtered.sort(function(a, b) {
    var an = a.user && a.user.fullname ? a.user.fullname : '';
    var bn = b.user && b.user.fullname ? b.user.fullname : '';
    if (v === 'name_asc')  return an.localeCompare(bn);
    if (v === 'name_desc') return bn.localeCompare(an);
    if (v === 'oldest')    return a.id - b.id;
    return b.id - a.id; /* newest */
  });
  page = 1;
  render();
}

  document.getElementById('submitBtn').textContent  = 'Save Changes';
  document.getElementById('f-name').value         = (s.user && s.user.fullname)      ? s.user.fullname      : '';
  document.getElementById('f-email').value        = (s.user && s.user.email)         ? s.user.email         : '';
  document.getElementById('f-phone_number').value = (s.user && s.user.phone_number)  ? s.user.phone_number  : '';
  document.getElementById('f-dob').value          = (s.user && s.user.date_of_birth) ? s.user.date_of_birth.slice(0,10) : '';
  document.getElementById('f-cin').value          = (s.user && s.user.cin)           ? s.user.cin           : '';
  document.getElementById('f-gender').value       = (s.user && s.user.gender)        ? s.user.gender        : '';
  document.getElementById('f-class').value        = (s.school_class && s.school_class.id) ? s.school_class.id : '';
  document.getElementById('f-teacher').value      = (s.school_class && s.school_class.teacher && s.school_class.teacher.user && s.school_class.teacher.user.fullname) ? s.school_class.teacher.user.fullname : '';
  document.getElementById('overlay').classList.add('open');
}
